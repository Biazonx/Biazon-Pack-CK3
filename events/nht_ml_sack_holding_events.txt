		add_gold = major_gold_value
		scope:nht_sack_previous_owner = {

				modifier = nht_sacked_holding_angry_opinion
				target = root

				years = 20
			}

		scope:nht_sacked_county = {
			add_county_modifier = {	#Prevent the county from being sacked again for quite a while.

				has_trait = just
