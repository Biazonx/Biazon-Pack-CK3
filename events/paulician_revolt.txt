namespace = paulician_revolt

paulician_revolt.0001 = {
    type = character_event
    title = paulician_revolt.0001.t
	desc = paulician_revolt.0001.desc
	override_background = { reference = throne_room }
	theme = emperor
	left_portrait = { character = root }

    option = {
        name = paulician_revolt.0001.a
			hidden_effect = { 
			title:e_byzantium = { 
			holder.top_liege = { save_scope_as = war_target }
				scope:war_target = {
					create_character = {
						name = "paulician_leader"
						gender = male
						culture = culture:greek
						faith = faith:paulician
						age = { 25 50 }
						stewardship = { 5 10 }
						prowess = { 5 15 }
						martial = { 5 15 }
						diplomacy = { 5 10 }
						learning = { 5 10 }
						trait = education_martial_3
						trait = ambitious
						trait = zealous
						trait = stubborn
						trait = heresiarch
						save_scope_as = rebel_leader
					}
				}
			title:c_satala = { save_scope_as = satala_county_title }
			title:c_satala = { set_county_faith = faith:paulician }
			title:c_satala = {
			random_county_province = {
					limit = { is_county_capital = yes }
					save_scope_as = rebellion_location
				}
			}
			title:c_azysia = { set_county_faith = faith:paulician }
			title:c_ablastha = { set_county_faith = faith:paulician }
			title:k_anatolia = { save_scope_as = anatolia_kingdom_title }

			scope:satala_county_title = {
				create_title_and_vassal_change = { type = granted save_scope_as = title_change_county }
				change_title_holder = { holder = scope:rebel_leader change = scope:title_change_county }
				resolve_title_and_vassal_change = scope:title_change_county
			}
			
			scope:rebel_leader = { add_gold = 500 add_piety = 1500 add_prestige = 1000  }
			scope:rebel_leader = {
	
					spawn_army = {
						location = scope:rebellion_location
						name = "paulician_army"
						men_at_arms = { type = skoutatoi men = 2500 }
						men_at_arms = { type = onager men = 50 }
						levies = 4000
						war_keep_on_attacker_victory = no
						inheritable = no
					}  
					start_war = {
						casus_belli = paulician_invasion
						target = scope:war_target
						target_title = scope:anatolia_kingdom_title
					}	
				}
			}
			clear_saved_scope = rebellion_location
			clear_saved_scope = war_target
			clear_saved_scope = rebel_leader
			clear_saved_scope = anatolia_kingdom_title
			clear_saved_scope = satala_county_title
			clear_saved_scope = title_change_county
		}
	}
	option = {
        name = paulician_revolt.0001.b
		hidden_effect = { 
			title:e_byzantium = { 
			holder.top_liege = { save_scope_as = war_target }
				scope:war_target = {
					create_character = {
						name = "paulician_leader"
						gender = male
						culture = culture:greek
						faith = faith:paulician
						age = { 25 50 }
						stewardship = { 5 10 }
						prowess = { 5 15 }
						martial = { 5 15 }
						diplomacy = { 5 10 }
						learning = { 5 10 }
						trait = education_martial_3
						trait = ambitious
						trait = zealous
						trait = stubborn
						trait = heresiarch
						save_scope_as = rebel_leader
					}
				}
			title:c_satala = { save_scope_as = satala_county_title }
			title:c_satala = { set_county_faith = faith:paulician }
			title:c_satala = {
			random_county_province = {
					limit = { is_county_capital = yes }
					save_scope_as = rebellion_location
				}
			}
			title:c_azysia = { set_county_faith = faith:paulician }
			title:c_ablastha = { set_county_faith = faith:paulician }
			title:k_anatolia = { save_scope_as = anatolia_kingdom_title }

			scope:satala_county_title = {
				create_title_and_vassal_change = { type = granted save_scope_as = title_change_county }
				change_title_holder = { holder = scope:rebel_leader change = scope:title_change_county }
				resolve_title_and_vassal_change = scope:title_change_county
			}
			
			scope:rebel_leader = { add_gold = 500 add_piety = 1500 add_prestige = 1000 }
			scope:rebel_leader = {
	
					spawn_army = {
						location = scope:rebellion_location
						name = "paulician_army"
						men_at_arms = { type = skoutatoi men = 2000 }
						men_at_arms = { type = onager men = 50 }
						levies = 4500
						war_keep_on_attacker_victory = no
						inheritable = no
					}  
					start_war = {
						casus_belli = paulician_invasion
						target = scope:war_target
						target_title = scope:anatolia_kingdom_title
					}	
				}
			}
			set_player_character = scope:rebel_leader
			clear_saved_scope = rebellion_location
			clear_saved_scope = war_target
			clear_saved_scope = anatolia_kingdom_title
			clear_saved_scope = satala_county_title
			clear_saved_scope = title_change_county
			clear_saved_scope = rebel_leader
		}
	}
}

paulician_revolt.1111 = {
	type = character_event
	hidden = yes
	title = paulician_revolt.1111.t
	desc = paulician_revolt.1111.desc
	override_background = { reference = throne_room }
	theme = emperor
	left_portrait = { character = root }
	option = {
		name = paulician_revolt.1111.a
		if = {
			limit = {
				has_game_rule = bgp_paulicians_on
			}

				id = paulician_revolt.0001
			}
		}
	}
}

	override_background = { reference = throne_room }
	theme = emperor
	left_portrait = { character = root }
	option = {
		name = paulician_revolt.0008.a	
	}
	option = {
		name = paulician_revolt.0008.b
		title:e_byzantium = { 
			holder.top_liege = { save_scope_as = byzantium_emperor}
		}
			add_gold = 500
			add_legitimacy = 150
			start_tributary = {
				contract_group = tributary_subjugated
				suzerain = scope:byzantium_emperor
			}
			tributary_contract_set_obligation_level = { type = default_tributary_taxes level = 1 }
			tributary_contract_set_obligation_level = { type = default_tributary_prestige level = 0 }
			tributary_contract_set_obligation_level = { type = suzerain_war_participation_guarantee level = 1 }
			tributary_contract_set_obligation_level = { type = tributary_war_participation_obligation level = 0 }
		}
	}
}
