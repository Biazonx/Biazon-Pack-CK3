weapon_inspiration_average_skill_value = {
	value = martial
	add = {
		value = prowess
		multiply = 2
	}
	divide = 2
	add = {	
		if = {
			limit = { 
				has_trait_xp = {
					trait = inspirational_blacksmith
					track = weaponsmith
					value >= 100
				}
			}
			value = 5
		}
		else_if = {
			limit = { 
				has_trait_xp = {
					trait = inspirational_blacksmith
					track = weaponsmith
					value >= 60
				}
			}
			value = 3
		}
		else_if = {
			limit = { 
				has_trait_xp = {
					trait = inspirational_blacksmith
					track = weaponsmith
					value >= 30
				}
			}
			value = 2
		}
		else_if = {
			limit = { has_trait = inspirational_blacksmith }
			value = 1
		}
	}
}
