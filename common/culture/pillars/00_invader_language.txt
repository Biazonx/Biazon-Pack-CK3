language_saiyan = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_saiyan
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_saiyan }
			multiply = 10
		}
	}
	
	color = saiyan
}
