bzn_blend_russian_traditions_decision = {
    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_culture.dds"
    }

    decision_group_type = regional_major
	
    ai_check_interval = 120

    cooldown = { years = 5 }

    is_shown = {
		NOT = {  # Can only do it once.
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:blend_russian_traditions
			}
		}
        is_landed = yes
		any_held_title = {
			tier = tier_county
			title_province = { geographical_region = world_siberia_2 }
        }
		has_character_modifier = russian_settlers
		has_character_modifier = russian_traders
    }
	
	is_valid = {
		has_character_modifier = russian_settlers
		has_character_modifier = russian_traders
	}

    desc = bzn_blend_russian_traditions_decision_desc

    effect = {
		culture:russian = { save_scope_as = original_russian_culture }
		culture = { save_scope_as = original_character_culture }
		create_hybrid_culture = culture:russian
		custom_tooltip = random_traditions_tt
		hidden_effect = {
			if = {
				limit = {
					root.culture = {
						NOT = {
							has_same_culture_heritage = scope:original_russian_culture
						}
					}
				}
				root.culture = {
					set_heritage_from = root.domicile.domicile_culture
				}
			}
			every_held_title = {
				limit = {
					tier = tier_county
					is_landless_type_title = no
				}
				set_county_culture = root.culture
			}
			# Spouses convert
			every_spouse = {
				limit = {
					is_courtier_of = root
					OR = {
						culture = scope:original_character_culture
						culture = scope:original_russian_culture
					}
					NOT = { culture = root.culture }
				}
				add_to_list = spouses_and_family_to_convert
			}
			# Family at court also convert
			every_close_or_extended_family_member = {
				limit = {
					is_courtier_of = root
					OR = {
						culture = scope:original_character_culture
						culture = scope:original_russian_culture
					}
					NOT = { culture = root.culture }
				}
				add_to_list = spouses_and_family_to_convert
			}
			every_in_list = {
				list = spouses_and_family_to_convert
				set_culture = root.culture
			}
			add_to_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:blend_russian_traditions
			}	
		}
	}

    cost = { prestige = 1000 }

    ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
