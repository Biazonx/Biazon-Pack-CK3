bzn_become_samurai = {
    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_knight_kneeling.dds"
    }
    decision_group_type = major
    ai_check_interval = 0

    is_shown = {
		culture = culture:japanese
        is_landless_adventurer = no
        NOT = { has_trait = samurai }
        NOT = { has_trait = ronin }
    }

    is_valid = {
        culture = culture:japanese
        prestige_level >= 3
		OR = {
        martial >= 12
        prowess >= 12
		}
        NOT = { has_trait = disloyal }
    }

    desc = bzn_become_samurai_desc
    cost = { prestige = 1000 }

    effect = {
        add_trait = samurai
    }

    ai_potential = {
		always = no
	}

	ai_will_do = {
		base = 0
	}
}

    decision_group_type = major

        add_trait = ronin

    ai_potential = {
		always = no
	}

		base = 0
	}
}
