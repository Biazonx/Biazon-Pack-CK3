wap_create_lower_niger_empire_decision = {
	picture = { reference = "gfx/interface/illustrations/decisions/decision_lower_niger_empire.dds" }

	desc = wap_create_lower_niger_empire_decision_desc
	selection_tooltip = wap_create_lower_niger_empire_decision_tooltip
	decision_group_type = regional_major

	ai_check_interval = 120
	cost = {
		gold = 500
		prestige = 1000
	}
	is_shown = {
		is_landed = yes
		#Must be in the right area.
		capital_province = { geographical_region = world_africa_west }
		#Can only do it once.
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:created_lower_niger_empire
			}
		}
	}

	is_valid = {
	    #Must be independent.
		is_independent_ruler = yes
		#Must be suitably prestigious.
		prestige_level >= high_prestige_level
		#Just confirm that becoming an empire will prevent you from achieving this one.
		highest_held_title_tier < tier_empire
		#Completely controls the core regions of their kingdoms.
		completely_controls_region = custom_lower_niger
	}

	is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
		is_at_war = no
	}

	effect = {
	
		save_scope_as = lower_niger_uniter
		gain_heroic_legend_seed_tooltip_effect = yes
		show_as_tooltip = { form_lower_niger_empire_scripted_effects = yes }
		
		trigger_event = wap_create_lower_niger_empire.0001
		hidden_effect = {
			#Can only be done once
			add_to_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:created_lower_niger_empire
			}
		}

	}
	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
